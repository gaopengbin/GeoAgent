```typescript
// stores/chatStore.ts
import { defineStore } from 'pinia'
import { useMapStore } from './mapStore'

interface Message {
  id: string
  role: 'user' | 'assistant'
  content: string
  timestamp: Date
  attachments?: FileAttachment[]
  toolSteps?: ToolStep[]
  mapCommands?: MapCommand[]
  charts?: ChartData[]
  report?: string
}

interface ToolStep {
  step: number
  tool_name: string
  tool_args?: Record<string, any>      // 工具调用参数
  description: string                  // 人类可读描述
  status: 'running' | 'done' | 'error' | 'retrying'
  // 以下字段在 tool_result 返回后填充
  success?: boolean
  summary?: string                     // 结果摘要，如"获取到326个地铁站"
  data_ref_id?: string                 // 结果数据引用ID
  preview?: {                          // 结果预览
    feature_count?: number
    geometry_type?: string
    bbox?: number[]
    sample_properties?: string[]
  }
  error?: string                       // 错误信息
  duration_ms?: number                 // 执行耗时
}

export const useChatStore = defineStore('chat', {
  state: () => ({
    sessionId: null as string | null,
    messages: [] as Message[],
    isStreaming: false,
    currentStep: null as ToolStep | null,
  }),
  
  actions: {
    async sendMessage(text: string, files: UploadedFile[]) {
      // 1. 添加用户消息
      this.messages.push({
        id: crypto.randomUUID(),
        role: 'user',
        content: text,
        timestamp: new Date(),
        attachments: files.map(f => ({ file_id: f.id, filename: f.name }))
      })
      
      // 2. 创建空的AI消息（流式填充）
      const aiMessage: Message = {
        id: crypto.randomUUID(),
        role: 'assistant',
        content: '',
        timestamp: new Date(),
        toolSteps: [],
        mapCommands: [],
        charts: [],
      }
      this.messages.push(aiMessage)
      this.isStreaming = true
      
      // 3. SSE流式请求
      const mapStore = useMapStore()
      
      const response = await fetch('/api/chat', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          session_id: this.sessionId,
          message: text,
          attachments: files.map(f => ({ file_id: f.id, filename: f.name }))
        })
      })
      
      const reader = response.body!.getReader()
      const decoder = new TextDecoder()
      
      while (true) {
        const { done, value } = await reader.read()
        if (done) break
        
        const chunk = decoder.decode(value)
        const lines = chunk.split('\n')
        
        for (const line of lines) {
          if (line.startsWith('data: ')) {
            const data = JSON.parse(line.slice(6))
            this.handleSSEEvent(aiMessage, data, mapStore)
          }
        }
      }
      
      this.isStreaming = false
    },
    
    handleSSEEvent(message: Message, data: any, mapStore: any) {
      switch (data.type) {
        case 'session':
          this.sessionId = data.session_id
          break
          
        case 'text':
          message.content += data.content
          break
          
        case 'thinking':
          // 纯文本思考过程（可选）
          break

        case 'tool_call':
          // Agent 决定调用工具 → 添加 running 状态步骤
          message.toolSteps!.push({
            step: data.step,
            tool_name: data.tool_name,
            tool_args: data.tool_args,
            description: data.description,
            status: 'running',
          })
          break

        case 'tool_result':
          // 工具执行完成 → 更新对应步骤的结果
          const existing = message.toolSteps!.find(s => s.step === data.step)
          if (existing) {
            existing.status = data.success ? 'done' : (data.retry ? 'retrying' : 'error')
            existing.success = data.success
            existing.summary = data.summary || data.error
            existing.data_ref_id = data.data_ref_id
            existing.preview = data.preview
            existing.error = data.error
            existing.duration_ms = data.duration_ms
          }
          break
          
        case 'map_command':
          message.mapCommands!.push(data.command)
          // 立即执行地图指令
          mapStore.executeCommand(data.command)
          break
          
        case 'chart':
          message.charts!.push(data.data)
          break
          
        case 'report':
          message.report = data.content
          break
          
        case 'done':
          this.sessionId = data.session_id
          break
      }
    }
  }
})
```

### 3.2 mapStore — 地图状态

```typescript
// stores/mapStore.ts
import { defineStore } from 'pinia'
import type { CommandExecutor } from '@/utils/commandExecutor'

interface MapLayer {
  id: string
  name: string
  type: 'geojson' | 'heatmap' | 'tileset' | 'track'
  visible: boolean
  data?: any
  style?: any
}

export const useMapStore = defineStore('map', {
  state: () => ({
    executor: null as CommandExecutor | null,
    layers: [] as MapLayer[],
    currentBasemap: 'tianditu_vec',
    loading: false,
    viewState: {
      longitude: 116.4,
      latitude: 39.9,
      height: 15000000,
    }
  }),
  
  actions: {
    setExecutor(exec: CommandExecutor) {
      this.executor = exec
    },
    
    executeCommand(command: MapCommand) {
      if (!this.executor) return
      
      this.executor.execute(command)
      
      // 同步图层状态
      if (command.action.includes('add') || command.action.includes('Layer')) {
        this.syncLayers()
      }
    },
    
    toggleLayer(layerId: string) {
      const layer = this.layers.find(l => l.id === layerId)
      if (layer) {
        layer.visible = !layer.visible
        this.executor?.execute({
          action: 'toggleLayer',
          params: { id: layerId, visible: layer.visible }
        })
      }
    },
    
    removeLayer(layerId: string) {
      this.layers = this.layers.filter(l => l.id !== layerId)
      this.executor?.execute({
        action: 'removeLayer',
        params: { id: layerId }
      })
    },
    
    zoomToLayer(layerId: string) {
      this.executor?.execute({
        action: 'zoomToLayer',
        params: { id: layerId }
      })
    },
    
    setBasemap(basemapId: string) {
      this.currentBasemap = basemapId
      this.executor?.execute({
        action: 'setBasemap',
        params: { type: basemapId }
      })
    }
  }
})
```

## 4. 指令执行引擎

```typescript
// utils/commandExecutor.ts
import * as Cesium from 'cesium'

interface MapCommand {
  action: string
  params: Record<string, any>
}

export class CommandExecutor {
  private viewer: Cesium.Viewer
  private layers: Map<string, any> = new Map()  // layerId → Cesium对象
  
  constructor(viewer: Cesium.Viewer) {
    this.viewer = viewer
  }
  
  execute(command: MapCommand) {
    const handler = this.getHandler(command.action)
    if (handler) {
      handler.call(this, command.params)
    } else {
      console.warn(`未知的地图指令: ${command.action}`)
    }
  }
  
  private getHandler(action: string) {
    const handlers: Record<string, Function> = {
      // 视图控制
      flyTo: this.handleFlyTo,
      zoomTo: this.handleZoomTo,
      resetView: this.handleResetView,
      
      // 图层操作
      addGeoJsonLayer: this.handleAddGeoJson,
      addHeatmap: this.handleAddHeatmap,
      addMarker: this.handleAddMarker,
      addPolyline: this.handleAddPolyline,
      removeLayer: this.handleRemoveLayer,
      toggleLayer: this.handleToggleLayer,
      zoomToLayer: this.handleZoomToLayer,
      
      // 动画
      playTrack: this.handlePlayTrack,
      
      // 底图
      setBasemap: this.handleSetBasemap,
    }
    return handlers[action]
  }
  
  // === 视图控制 ===
  
  private handleFlyTo(params: any) {
    this.viewer.camera.flyTo({
      destination: Cesium.Cartesian3.fromDegrees(
        params.longitude, 
        params.latitude, 
        params.height || 10000
      ),
      orientation: {
        heading: Cesium.Math.toRadians(params.heading || 0),
        pitch: Cesium.Math.toRadians(params.pitch || -90),
        roll: 0
      },
      duration: params.duration || 2
    })
  }
  
  // === 图层操作 ===
  
  private handleAddGeoJson(params: any) {
    const dataSource = Cesium.GeoJsonDataSource.load(params.data, {
      stroke: Cesium.Color.fromCssColorString(params.style?.color || '#3388ff'),
      strokeWidth: params.style?.strokeWidth || 2,
      fill: Cesium.Color.fromCssColorString(params.style?.fillColor || '#3388ff').withAlpha(params.style?.opacity || 0.3),
      markerSize: params.style?.size || 12,
    })
    
    dataSource.then((ds: Cesium.GeoJsonDataSource) => {
      ds.name = params.name || params.id
      this.viewer.dataSources.add(ds)
      this.layers.set(params.id, ds)
      
      // 自动飞到数据范围
      this.viewer.flyTo(ds, { duration: 1.5 })
    })
  }
  
  private handleAddHeatmap(params: any) {
    // 使用 CesiumHeatmap 或自定义 Primitive
    // 将热力图点数据渲染为 heatmap overlay
    const points = params.points.map((p: any) => ({
      x: p.longitude,
      y: p.latitude,
      value: p.weight
    }))
    
    // 创建热力图图层（具体实现依赖cesium-heatmap插件或自定义Entity方式）
    // ...
    this.layers.set(params.id, heatmapLayer)
  }
  
  private handleRemoveLayer(params: any) {
    const layer = this.layers.get(params.id)
    if (layer) {
      if (layer instanceof Cesium.GeoJsonDataSource) {
        this.viewer.dataSources.remove(layer, true)
      }
      this.layers.delete(params.id)
    }
  }
  
  private handleToggleLayer(params: any) {
    const layer = this.layers.get(params.id)
    if (layer) {
      layer.show = params.visible
    }
  }
  
  // === 轨迹动画 ===
  
  private handlePlayTrack(params: any) {
    const positions = params.positions
    const property = new Cesium.SampledPositionProperty()
    
    const startTime = Cesium.JulianDate.fromIso8601(positions[0].time)
    
    positions.forEach((p: any) => {
      const time = Cesium.JulianDate.fromIso8601(p.time)
      const pos = Cesium.Cartesian3.fromDegrees(p.longitude, p.latitude, p.altitude || 0)
      property.addSample(time, pos)
    })
    
    const entity = this.viewer.entities.add({
      position: property,
      point: { pixelSize: 10, color: Cesium.Color.RED },
      path: { 
        width: 3, 
        material: Cesium.Color.YELLOW,
        leadTime: 0,
        trailTime: 3600
      }
    })
    
    // 设置时间轴
    const endTime = Cesium.JulianDate.fromIso8601(positions[positions.length - 1].time)
    this.viewer.clock.startTime = startTime
    this.viewer.clock.stopTime = endTime
    this.viewer.clock.currentTime = startTime
    this.viewer.clock.multiplier = params.speed || 10
    this.viewer.clock.shouldAnimate = true
    
    this.viewer.trackedEntity = entity
    this.layers.set(params.trackId || 'track', entity)
  }
}
```
